22inline bool isDarkMode()
23{
24 const QPalette defaultPalette;
25 const auto text = defaultPalette.color(QPalette::WindowText);
26 const auto window = defaultPalette.color(QPalette::Window);
27 return text.lightness() > window.lightness();
28}

43inline QStringList checkSubDir(const QString &path)
44{
45 QStringList paths;
46 auto dir = QDir(path);
47 const QFileInfoList items = dir.entryInfoList({"*"}, QDir::Dirs | QDir::Files | QDir::NoDotAndDotDot);
48 for (const QFileInfo &item : items) {
49 if (item.isDir()) {
50 paths.append(item.absoluteFilePath());
51 paths.append(checkSubDir(item.absoluteFilePath()));
52 }
53 }
54 return paths;
55}
