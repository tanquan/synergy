146// enter screen: primary -> secondary
147// entering screen at screen position $1 = x, $2 = y. x,y are
148// absolute screen coordinates. $3 = sequence number, which is
149// used to order messages between screens. the secondary screen
150// must return this number with some messages. $4 = modifier key
151// mask. this will have bits set for each toggle modifier key
152// that is activated on entry to the screen. the secondary screen
153// should adjust its toggle modifiers to reflect that state.
154extern const char *const kMsgCEnter;
155
156// leave screen: primary -> secondary
157// leaving screen. the secondary screen should send clipboard
158// data in response to this message for those clipboards that
159// it has grabbed (i.e. has sent a kMsgCClipboard for and has
160// not received a kMsgCClipboard for with a greater sequence
161// number) and that were grabbed or have changed since the
162// last leave.
163extern const char *const kMsgCLeave;
164
165// grab clipboard: primary <-> secondary
166// sent by screen when some other app on that screen grabs a
167// clipboard. $1 = the clipboard identifier, $2 = sequence number.
168// secondary screens must use the sequence number passed in the
169// most recent kMsgCEnter. the primary always sends 0.
170extern const char *const kMsgCClipboard;

172// screensaver change: primary -> secondary

180// resolution change acknowledgment: primary -> secondary

186// keep connection alive: primary <-> secondary
187// sent by the server periodically to verify that connections are still
188// up and running. clients must reply in kind on receipt. if the server
189// gets an error sending the message or does not receive a reply within
190// a reasonable time then the server disconnects the client. if the
191// client doesn't receive these (or any message) periodically then it
192// should disconnect from the server. the appropriate interval is
193// defined by an option.
194extern const char *const kMsgCKeepAlive;

203// key pressed: primary -> secondary
204// $1 = KeyID, $2 = KeyModifierMask, $3 = KeyButton
205// the KeyButton identifies the physical key on the primary used to
206// generate this key. the secondary should note the KeyButton along
207// with the physical key it uses to generate the key press. on
208// release, the secondary can then use the primary's KeyButton to
209// find its corresponding physical key and release it. this is
210// necessary because the KeyID on release may not be the KeyID of
211// the press. this can happen with combining (dead) keys or if
212// the keyboard layouts are not identical and the user releases
213// a modifier key before releasing the modified key.
214// languageCode is parameter which helps client to react on unknwon
215// language letters
216extern const char *const kMsgDKeyDown;
